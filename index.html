precision mediump float; varying vec3 normal_out; varying vec2 uvs_out; varying vec3 lightPositionOut; void main() { vec3 lightDir = lightPositionOut; float d = max(dot(normal_out, lightDir), 0.0); vec3 color = vec3(0.1, 0.1, 0.1) + vec3(1, 1, 1) * d; gl_FragColor = vec4(color, 1.0); }
